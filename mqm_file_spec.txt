// this is a spec for the map format
//
// there are still unimplemented features
// max map dimensions are 128x128 for now

s : start
f : finish

# : solid wall
^ : elevated floor
v : low ceiling
x : hole in the ground (player can fall to death)
- : bridge, platform in between floor and ceiling
% : destroyable wall


a : ammo
h : health

r : rocket launcher
m : machine gun
g : shotgun

w : walking enemy
t : turret enemy
